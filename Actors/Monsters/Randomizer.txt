Actor BaseSpawner
{
	+IsMonster
	+Shootable
	+NoBlockMap
	+ThruActors
	+NoInterAction
	+DontBlast
	+Notarget
	+NoPain
	+NoDamage
	+NOTONAUTOMAP
	+NEVERRESPAWN
	+SERVERSIDEONLY
	Health 2147483647
	Mass 2147483647
	Speed 0
	var int user_tid;
	var int user_class;
}

ACTOR MancubiRandomizer : BaseSpawner Replaces Fatso
{
	Scale 0.6 
	RenderStyle "Translucent"
	+BossDeath
States
    {
	 Spawn:
		TNT1 A 0
		TNT1 A 0 ACS_NamedExecuteAlways("CFzA_DisableCryocubus", 0, 0, 0, 0)
		TNT1 A 5
		TNT1 A 0 A_Jump(128, "Mancubus")
		TNT1 A 0 A_Jump(128, "Cryocubus")
		//TNT1 A 0 A_Jump(128, "Volcabus")
	 Mancubus:
		TNT1 A 0 A_SpawnItemEx ("NewMancubus",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION | SXF_TRANSFERAMBUSHFLAG | SXF_TRANSFERPOINTERS ,0, ACS_NamedExecuteWithResult("SetUserTID"))
		Goto Check
	Cryocubus:
		TNT1 A 0 A_SpawnItemEx ("Cryocubus",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION | SXF_TRANSFERAMBUSHFLAG | SXF_TRANSFERPOINTERS ,0, ACS_NamedExecuteWithResult("SetUserTID"))
		Goto Check
	/* Volcabus:
		TNT1 A 0 A_SpawnItemEx ("NewVolcabus",0,0,0,0,0,0,0,SXF_NOCHECKPOSITION | SXF_TRANSFERAMBUSHFLAG | SXF_TRANSFERPOINTERS ,0, ACS_NamedExecuteWithResult("SetUserTID"))
		Goto Check*/
	Check:
		TNT1 A 5 A_JumpIf(ACS_NamedExecuteWithResult("Live_Check", user_tid) != 1, "Execute")
		TNT1 A 5 ACS_NamedExecuteAlways("Get_Spawner_Status", 0, user_tid)
		Loop
	Execute:
		TNT1 A 2 A_Die
		Goto Death
	Death:
		TNT1 A 5 A_Fall
		TNT1 A 2 A_BossDeath
		Stop	
	}
}
